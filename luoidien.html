async function runAnalysis(type) { const tuyen = document.getElementById('selectTuyen').value; if (!tuyen) return alert("Chọn tuyến dây trước!"); toggleSidebar(); const loading = document.getElementById('loading'); const log = document.getElementById('liveLog'); const bar = document.getElementById('myBar'); const label = document.getElementById('pLabel'); const body = document.getElementById('reportBody'); loading.style.display = 'flex'; log.innerHTML = `
> Bắt đầu quét an toàn cho Tuyến: ${tuyen}
`; body.innerHTML = ""; const pts = masterData[tuyen].points; let findings = []; for (let i = 0; i < pts.length - 1; i++) { let pct = Math.round(((i + 1) / (pts.length - 1)) * 100); bar.style.width = pct + '%'; label.innerText = `Đang quét: ${pct}%`; const midLat = (pts[i].lat + pts[i+1].lat) / 2; const midLng = (pts[i].lng + pts[i+1].lng) / 2; let logMsg = document.createElement('div'); logMsg.innerHTML = `> Đang kiểm tra khoảng Trụ ${pts[i].tru}-${pts[i+1].tru}...`; log.appendChild(logMsg); log.scrollTop = log.scrollHeight; let success = false; let retryCount = 0; // Cơ chế thử lại nếu bị lỗi kết nối while (!success && retryCount < 3) { try { // TĂNG THỜI GIAN NGHỈ LÊN 1.2 GIÂY ĐỂ TRÁNH BỊ CHẶN API await new Promise(r => setTimeout(r, 1200)); const res = await fetch(`https://nominatim.openstreetmap.org/reverse?format=json&lat=${midLat}&lon=${midLng}&zoom=18`, { headers: { 'User-Agent': 'VDC_Safety_App_v10' } }); if (res.status === 429) { // Lỗi Too Many Requests throw new Error("API Limit"); } const data = await res.json(); if (data && data.address) { const addr = data.address; let objName = (type === 'road') ? (addr.road || addr.highway || addr.suburb || "") : (addr.waterway || addr.river || addr.canal || addr.stream || ""); if (objName) { let resultLog = document.createElement('div'); resultLog.style.color = "#ffff00"; resultLog.innerHTML = `[!] PHÁT HIỆN GIAO CHÉO: ${objName}`; log.appendChild(resultLog); findings.push({ stt: findings.length + 1, code: tuyen, span: `Trụ ${pts[i].tru} ↔ ${pts[i+1].tru}`, name: objName, area: (addr.suburb || addr.quarter || addr.town || "TP.HCM"), coord: `${midLat.toFixed(6)}, ${midLng.toFixed(6)}` }); } } success = true; // Quét thành công điểm này } catch (e) { retryCount++; let errMsg = document.createElement('div'); errMsg.style.color = "#ff9800"; // Màu cam cảnh báo thử lại errMsg.innerHTML = `> API quá tải, đang thử lại lần ${retryCount}...`; log.appendChild(errMsg); await new Promise(r => setTimeout(r, 2000)); // Chờ 2 giây rồi mới thử lại } } } // Đổ dữ liệu vào bảng sau khi kết thúc if (findings.length === 0) { body.innerHTML = "Không tìm thấy giao cắt nào. Hãy kiểm tra lại loại hình Quét (Đường/Sông)."; } else { findings.forEach(d => { body.innerHTML += `${d.stt}${d.code}${d.span}${d.name}${d.area}${d.coord}`; }); } loading.style.display = 'none'; document.getElementById('resultPanel').classList.add('active'); document.getElementById('resTitle').innerText = `KẾT QUẢ QUÉT ${tuyen} - ${type.toUpperCase()}`; }
